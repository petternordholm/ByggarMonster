package ${packageName};

public class ${className}Builder {
{EACH members}    private ${type} ${name};
{/EACH}
    public ${className} build() {
        return new ${className}({EACH constructorParameters ", "}${name}{/EACH});
    }
{EACH members}
    public ${className}Builder with${name}(final ${type} ${name}) {
        this.${name} = ${name};
        return this;
    }
{/EACH}
}
